// Bridge node id = `bridge:${commandName}` (RESEARCH §Pitfall 6 — avoid collision with file paths).

  fy: 0,                         // D-13 — pin to boundary line
  // fx set below via alphabetic x-spread
}));

// Alphabetic x-spread — RESEARCH §Pattern 5 hash-gated cache.
const sortedBridges = [...bridgeNodes].sort((a, b) => a.commandName!.localeCompare(b.commandName!));
const bridgeSetHash = sortedBridges.map((n) => n.commandName).join(',');
const prevHash = get().lastBridgeSetHash;
if (bridgeSetHash !== prevHash || sortedBridges.length === 0) {
  // Compute new fx values. N bridges → evenly spread across [-GRAPH_HALF_WIDTH, +GRAPH_HALF_WIDTH].
  const N = sortedBridges.length;
  sortedBridges.forEach((n, i) => {
    n.fx = N === 1 ? 0 : -GRAPH_HALF_WIDTH + (2 * GRAPH_HALF_WIDTH * i) / (N - 1);
  });
  set({ lastBridgeSetHash: bridgeSetHash });
} else {
  // Same command set — preserve existing fx values from prior snapshot.
  const priorById = new Map(get().graphNodes.filter((n) => n.kind === 'bridge').map((n) => [n.id, n.fx]));
  sortedBridges.forEach((n) => { n.fx = priorById.get(n.id) ?? 0; });
}

// Language classification for file nodes (D-16) — derived on the fly.
// Apply to existing fileNodes array BEFORE appending bridgeNodes.
const classifyLanguage = (path: string): 'ts' | 'rust' | undefined => {
  if (path.startsWith('src-tauri/')) return 'rust';
  if (path.endsWith('.rs')) return 'rust';
  if (/\.(ts|tsx|js|jsx)$/.test(path)) return 'ts';
  return undefined;
};
// fileNodes: apply kind='file' + language to each
for (const n of fileNodes) {
  n.kind = 'file';
  n.language = classifyLanguage(n.id);
}

// Merge: files first, then bridges (bridges drawn on top in RadarCanvas z-order anyway).
const allNodes = [...fileNodes, ...sortedBridges];

// Bridge edges — fan invokes from callers to bridge, single handles from bridge to handler.
const bridgeEdges: GraphEdge[] = [];
for (const b of bridges) {
  const bridgeId = `bridge:${b.commandName}`;
  for (const caller of b.callerFiles) {
    bridgeEdges.push({ source: caller.file, target: bridgeId, kind: 'invokes' as EdgeKind });
  }
  if (b.handlerFile) {
    bridgeEdges.push({ source: bridgeId, target: b.handlerFile, kind: 'handles' as EdgeKind });
  }
}

// Final: graphNodes = allNodes; graphEdges = [...depEdges, ...bridgeEdges].
set({ graphNodes: allNodes, graphEdges: [...depEdges, ...bridgeEdges] });

```typescript
import type { SimulationNodeDatum } from 'd3-force';

export interface BoundaryNode extends SimulationNodeDatum {
  kind: 'file' | 'bridge';
  language?: 'ts' | 'rust';
}

export interface BoundaryForce {
  (alpha: number): void;
  initialize: (nodes: BoundaryNode[]) => void;
  strength: ((v: number) => BoundaryForce) & (() => number);
}

export const BOUNDARY_TARGET_Y_MAGNITUDE = 300;
export const BOUNDARY_DEADBAND = 5;
export const FORCE_BOUNDARY_BASE_STRENGTH = 0.15;

export function forceBoundary(): BoundaryForce {
  let nodes: BoundaryNode[] = [];
  let strength = FORCE_BOUNDARY_BASE_STRENGTH;

  const force = ((alpha: number) => {
    const k = strength * alpha;
    if (k === 0) return;    // Pitfall 7 — zero-strength early-return

    for (const n of nodes) {
      // D-10: bridges skipped (fx/fy-pinned); only files feel the force.
      if (n.kind !== 'file') continue;
      // D-16: files without ts/rust language classification skip.
      if (n.language === undefined) continue;

      const targetY = n.language === 'ts' ? -BOUNDARY_TARGET_Y_MAGNITUDE : BOUNDARY_TARGET_Y_MAGNITUDE;
      const y = n.y ?? 0;
      const dy = targetY - y;
      // Deadband: if near y=0 and far from target, don't correct (prevents jitter across the line).
      if (Math.abs(y) < BOUNDARY_DEADBAND && Math.abs(dy) > (BOUNDARY_TARGET_Y_MAGNITUDE - BOUNDARY_DEADBAND)) continue;
      n.vy = (n.vy ?? 0) + Math.sign(dy) * k * Math.min(Math.abs(dy), BOUNDARY_TARGET_Y_MAGNITUDE);
    }
  }) as BoundaryForce;

  force.initialize = (n: BoundaryNode[]) => { nodes = n; };
  force.strength = ((v?: number) => {
    if (v === undefined) return strength;
    strength = v;
    return force;
  }) as BoundaryForce['strength'];
  return force;
}
```

function mulberry32(seed: number): () => number {
  let t = seed;
  return () => {
    t = (t + 0x6d2b79f5) | 0;
    let r = Math.imul(t ^ (t >>> 15), 1 | t);
    r = (r + Math.imul(r ^ (r >>> 7), 61 | r)) ^ r;
    return ((r ^ (r >>> 14)) >>> 0) / 4294967296;
  };
}

function step(nodes: BoundaryNode[]) {

    n.vy = (n.vy ?? 0) * 0.5;
    n.x = (n.x ?? 0) + (n.vx ?? 0);
    n.y = (n.y ?? 0) + (n.vy ?? 0);
  }
}
