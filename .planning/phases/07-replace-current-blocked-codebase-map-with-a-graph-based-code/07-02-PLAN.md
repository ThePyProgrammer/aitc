**Step 1 — extract.rs.** Implement `extract_imports(tree, source, language) -> Vec<RawImport>`. Pattern:

```rust
use tree_sitter::{Parser, Query, QueryCursor, StreamingIterator};
use crate::pipeline::deps::queries::{typescript::TYPESCRIPT_IMPORTS, /* etc */};
use crate::pipeline::deps::EdgeKind;
use std::time::{Duration, Instant};

pub const MAX_FILE_SIZE_BYTES: u64 = 1_048_576; // 1MB — T-07-A
pub const MAX_PARSE_DURATION: Duration = Duration::from_millis(500); // T-07-A

pub fn parse_and_extract(
    path: &std::path::Path,
    language: SourceLanguage,
) -> Vec<RawImport> {
    // T-07-A: file size cap
    let metadata = match std::fs::metadata(path) {
        Ok(m) => m,
        Err(_) => return Vec::new(),
    };
    if metadata.len() > MAX_FILE_SIZE_BYTES {
        tracing::trace!("dep_graph: skipping oversize file {} ({}B)", path.display(), metadata.len());
        return Vec::new();
    }
    let source = match std::fs::read_to_string(path) {
        Ok(s) => s,
        Err(_) => return Vec::new(),
    };
    let mut parser = Parser::new();
    let ts_lang = ts_language_for(language);
    if parser.set_language(&ts_lang).is_err() {
        return Vec::new();
    }
    // T-07-A: parse timeout (tree-sitter has built-in via set_timeout_micros)
    parser.set_timeout_micros(MAX_PARSE_DURATION.as_micros() as u64);
    let started = Instant::now();
    let tree = match parser.parse(&source, None) {
        Some(t) => t,
        None => {
            tracing::trace!("dep_graph: parse timeout/failure {}", path.display());
            return Vec::new();
        }
    };
    if started.elapsed() > MAX_PARSE_DURATION {
        return Vec::new();
    }
    extract_imports(&tree, &source, language)
}

fn ts_language_for(lang: SourceLanguage) -> tree_sitter::Language {
    match lang {
        SourceLanguage::TypeScript => tree_sitter_typescript::LANGUAGE_TYPESCRIPT.into(),
        SourceLanguage::Tsx => tree_sitter_typescript::LANGUAGE_TSX.into(),
        SourceLanguage::JavaScript => tree_sitter_javascript::LANGUAGE.into(),
        SourceLanguage::Jsx => tree_sitter_javascript::LANGUAGE.into(), // JSX uses JS grammar
        SourceLanguage::Rust => tree_sitter_rust::LANGUAGE.into(),
        SourceLanguage::Python => tree_sitter_python::LANGUAGE.into(),
    }
}

pub fn extract_imports(
    tree: &tree_sitter::Tree,
    source: &str,
    language: SourceLanguage,
) -> Vec<RawImport> {
    let (lang_obj, query_str) = match language {
        SourceLanguage::TypeScript | SourceLanguage::Tsx => {
            (ts_language_for(language), TYPESCRIPT_IMPORTS)
        }
        SourceLanguage::JavaScript | SourceLanguage::Jsx => {
            (ts_language_for(language), crate::pipeline::deps::queries::javascript::JAVASCRIPT_IMPORTS)
        }
        SourceLanguage::Rust => {
            (ts_language_for(language), crate::pipeline::deps::queries::rust::RUST_IMPORTS)
        }
        SourceLanguage::Python => {
            (ts_language_for(language), crate::pipeline::deps::queries::python::PYTHON_IMPORTS)
        }
    };
    let query = match Query::new(&lang_obj, query_str) {
        Ok(q) => q,
        Err(e) => {
            tracing::error!("dep_graph: query compile failed: {e:?}");
            return Vec::new();
        }
    };
    let mut cursor = QueryCursor::new();
    let mut out = Vec::new();
    let mut matches = cursor.matches(&query, tree.root_node(), source.as_bytes());
    while let Some(m) = matches.next() {
        // Map capture name → kind. The pattern indices follow the order in the
        // S-expression query, so for TS/JS: 0 = import, 1 = reexport, 2 = dynamic,
        // 3 = require. For Rust: 0 = use, 1 = mod. For Python: 0 = import, 1 = from.
        let kind = match (language, m.pattern_index) {
            (SourceLanguage::TypeScript | SourceLanguage::Tsx | SourceLanguage::JavaScript | SourceLanguage::Jsx, 0) => {
                // detect `import type { ... } from ...`: check the import_statement node text starts with "import type"
                let node_text = m.captures.iter()
                    .find(|c| query.capture_names()[c.index as usize] == "import")
                    .and_then(|c| c.node.utf8_text(source.as_bytes()).ok())
                    .unwrap_or("");
                if node_text.trim_start().starts_with("import type") {
                    EdgeKind::TypeOnly
                } else {
                    EdgeKind::Import
                }
            }
            (SourceLanguage::TypeScript | SourceLanguage::Tsx | SourceLanguage::JavaScript | SourceLanguage::Jsx, 1) => EdgeKind::Reexport,
            (SourceLanguage::TypeScript | SourceLanguage::Tsx | SourceLanguage::JavaScript | SourceLanguage::Jsx, 2 | 3) => EdgeKind::DynamicImport,
            (SourceLanguage::Rust, 0) => EdgeKind::Use,
            (SourceLanguage::Rust, 1) => EdgeKind::ModDecl,
            (SourceLanguage::Python, 0) => EdgeKind::ImportStmt,
            (SourceLanguage::Python, 1) => EdgeKind::FromImport,
            _ => continue,
        };
        // Find the @path or @name capture
        for c in m.captures {
            let name = &query.capture_names()[c.index as usize];
            if *name == "path" || *name == "name" {
                if let Ok(spec) = c.node.utf8_text(source.as_bytes()) {
                    out.push(RawImport { spec: spec.to_string(), kind: kind.clone() });
                }
            }
        }
    }
    out
}
```

**Step 2 — resolve.rs.** Implement per-language resolvers + master `resolve_import` dispatch:

```rust
use std::path::{Path, PathBuf};

pub const TS_EXTENSIONS: &[&str] = &["ts", "tsx", "js", "jsx", "mts", "cts", "mjs", "cjs"];
pub const RUST_EXTENSIONS: &[&str] = &["rs"];
pub const PYTHON_EXTENSIONS: &[&str] = &["py"];

#[derive(Debug, Clone, Default)]
pub struct ResolveContext {
    pub tsconfig_paths: Vec<(String, Vec<PathBuf>)>, // alias, target dirs (repo-relative)
}

pub fn resolve_import(
    spec: &str,
    from_file: &Path,
    repo_root: &Path,
    language: crate::pipeline::deps::extract::SourceLanguage,
    ctx: &ResolveContext,
) -> Option<PathBuf> {
    use crate::pipeline::deps::extract::SourceLanguage::*;
    let resolved = match language {
        TypeScript | Tsx | JavaScript | Jsx => resolve_ts_import(spec, from_file, repo_root, &ctx.tsconfig_paths),
        Rust => resolve_rust_import(spec, from_file, repo_root),
        Python => resolve_python_import(spec, from_file, repo_root),
    }?;
    // T-07-B: path-traversal check. Canonicalize and verify inside repo_root.
    let canonical = resolved.canonicalize().ok()?;
    let canonical_root = repo_root.canonicalize().ok()?;
    if !canonical.starts_with(&canonical_root) {
        tracing::trace!("dep_graph: path traversal attempt {} -> {}", spec, canonical.display());
        return None;
    }
    Some(canonical)
}

pub fn resolve_ts_import(
    spec: &str,
    from_file: &Path,
    repo_root: &Path,
    tsconfig_paths: &[(String, Vec<PathBuf>)],
) -> Option<PathBuf> {
    if spec.starts_with("./") || spec.starts_with("../") {
        let base = from_file.parent()?;
        return resolve_with_extensions(&base.join(spec), TS_EXTENSIONS);
    }
    for (alias, targets) in tsconfig_paths {
        let stripped = alias.trim_end_matches('*');
        if let Some(rest) = spec.strip_prefix(stripped) {
            for target in targets {
                let target_clean = target.to_string_lossy().trim_end_matches('*').to_string();
                let candidate = repo_root.join(&target_clean).join(rest.trim_start_matches('/'));
                if let Some(p) = resolve_with_extensions(&candidate, TS_EXTENSIONS) {
                    return Some(p);
                }
            }
        }
    }
    None // bare specifier — D-07 external, dropped
}

pub fn resolve_rust_import(
    spec: &str,
    from_file: &Path,
    repo_root: &Path,
) -> Option<PathBuf> {
    // Rust spec is like `crate::foo::Bar` or `super::baz` or `mod sibling;`.
    // For `mod sibling;` the spec is just "sibling" — resolve relative to from_file.parent().
    // For `crate::foo::Bar` resolve as repo_root/<crate-root>/foo.rs (best-effort: drop the last segment as it's the item).
    let segments: Vec<&str> = spec.split("::").collect();
    if segments.is_empty() { return None; }
    let path_segments: Vec<&str> = if segments[0] == "crate" {
        segments[1..].to_vec()
    } else if segments[0] == "self" || segments[0] == "super" {
        // Relative — resolve from from_file's directory walking up
        return resolve_rust_relative(&segments, from_file);
    } else if segments.len() == 1 {
        // `mod sibling;` or `use sibling;` — sibling file/dir relative to from_file
        return resolve_rust_relative(&segments, from_file);
    } else {
        // External crate (std, serde, etc.) — D-07 drop
        return None;
    };
    // Drop the last segment (item being imported, not a file)
    let head = if path_segments.len() > 1 { &path_segments[..path_segments.len() - 1] } else { &path_segments[..] };
    if head.is_empty() { return None; }
    // Try repo_root + segments + ".rs", or .../mod.rs
    let mut candidate = repo_root.to_path_buf();
    candidate.push("src"); // best-effort: assume crate root is src/
    for s in head { candidate.push(s); }
    resolve_with_extensions(&candidate, RUST_EXTENSIONS)
        .or_else(|| {
            let mod_rs = candidate.join("mod.rs");
            if mod_rs.is_file() { Some(mod_rs) } else { None }
        })
}

fn resolve_rust_relative(segments: &[&str], from_file: &Path) -> Option<PathBuf> {
    let mut base = from_file.parent()?.to_path_buf();
    let mut iter = segments.iter().peekable();
    while let Some(s) = iter.peek() {
        if **s == "super" { base = base.parent()?.to_path_buf(); iter.next(); }
        else if **s == "self" { iter.next(); }
        else { break; }
    }
    let rest: Vec<&&str> = iter.collect();
    if rest.is_empty() { return None; }
    let head = if rest.len() > 1 { &rest[..rest.len() - 1] } else { &rest[..] };
    let mut candidate = base;
    for s in head { candidate.push(s); }
    resolve_with_extensions(&candidate, RUST_EXTENSIONS)
        .or_else(|| {
            let mod_rs = candidate.join("mod.rs");
            if mod_rs.is_file() { Some(mod_rs) } else { None }
        })
}

pub fn resolve_python_import(
    spec: &str,
    from_file: &Path,
    repo_root: &Path,
) -> Option<PathBuf> {
    if spec.starts_with('.') {
        // Relative: count leading dots = how many parents to walk up
        let dots = spec.chars().take_while(|c| *c == '.').count();
        let rest = &spec[dots..];
        let mut base = from_file.parent()?.to_path_buf();
        for _ in 1..dots { base = base.parent()?.to_path_buf(); }
        let parts: Vec<&str> = rest.split('.').filter(|s| !s.is_empty()).collect();
        for p in &parts { base.push(p); }
        return resolve_python_module(&base, repo_root);
    }
    // Absolute python import: try repo_root/<dotted/path>
    let parts: Vec<&str> = spec.split('.').collect();
    let mut candidate = repo_root.to_path_buf();
    for p in &parts { candidate.push(p); }
    resolve_python_module(&candidate, repo_root)
}

fn resolve_python_module(candidate: &Path, _repo_root: &Path) -> Option<PathBuf> {
    let py = candidate.with_extension("py");
    if py.is_file() { return Some(py); }
    let init = candidate.join("__init__.py");
    if init.is_file() { return Some(init); }
    None
}

pub fn resolve_with_extensions(path: &Path, extensions: &[&str]) -> Option<PathBuf> {
    if path.is_file() { return Some(path.to_path_buf()); }
    for ext in extensions {
        let candidate = path.with_extension(ext);
        if candidate.is_file() { return Some(candidate); }
    }
    if path.is_dir() {
        for ext in extensions {
            let candidate = path.join(format!("index.{ext}"));
            if candidate.is_file() { return Some(candidate); }
        }
    }
    None
}
```

**Step 1 — Update `mod.rs` to add caps + return type.** Change return type to include degraded flag:

```rust
pub const MAX_EDGES_PER_NODE: usize = 200;        // T-07-C
pub const MAX_TOTAL_EDGES: usize = 100_000;       // T-07-C

#[derive(Debug, Clone)]
pub struct DependencyGraphResult {
    pub edges: Vec<DependencyEdge>,
    pub degraded: bool,
    pub unresolved_count: usize,  // for the UI-SPEC "{N}_IMPORTS_UNRESOLVED" pill
}

pub fn build_dependency_graph(
    repo_root: &std::path::Path,
    files: &[std::path::PathBuf],
) -> DependencyGraphResult {
    use rayon::prelude::*;
    use crate::pipeline::deps::extract::{parse_and_extract, detect_language};
    use crate::pipeline::deps::resolve::{resolve_import, ResolveContext};
    use std::sync::atomic::{AtomicUsize, Ordering};

    let ctx = ResolveContext::default(); // TODO Plan 03+ pass tsconfig — for now empty
    let unresolved = AtomicUsize::new(0);

    let per_file_edges: Vec<Vec<DependencyEdge>> = files
        .par_iter()
        .map(|path| {
            let lang = match detect_language(path) {
                Some(l) => l,
                None => return Vec::new(),
            };
            let raw = parse_and_extract(path, lang);
            let mut out = Vec::with_capacity(raw.len().min(MAX_EDGES_PER_NODE));
            for r in raw {
                if out.len() >= MAX_EDGES_PER_NODE {
                    tracing::warn!(
                        "dep_graph: per-node edge cap hit for {} ({} imports)",
                        path.display(), out.len()
                    );
                    break;
                }
                match resolve_import(&r.spec, path, repo_root, lang, &ctx) {
                    Some(target) => out.push(DependencyEdge {
                        from: path.clone(),
                        to: target,
                        kind: r.kind,
                    }),
                    None => { unresolved.fetch_add(1, Ordering::Relaxed); }
                }
            }
            out
        })
        .collect();

    let mut all: Vec<DependencyEdge> = Vec::new();
    let mut degraded = false;
    for batch in per_file_edges {
        if all.len() + batch.len() > MAX_TOTAL_EDGES {
            let take = MAX_TOTAL_EDGES.saturating_sub(all.len());
            all.extend(batch.into_iter().take(take));
            degraded = true;
            tracing::warn!("dep_graph: total edge cap hit ({}); truncating", MAX_TOTAL_EDGES);
            break;
        }
        all.extend(batch);
    }

    DependencyGraphResult {
        edges: all,
        degraded,
        unresolved_count: unresolved.load(Ordering::Relaxed),
    }
}
```
