// D-14, D-15, D-16, D-17, D-18, VIZN-02.
// Pure functions for comet trail lifecycle and rendering.
// Render functions called from RadarCanvas's rAF loop.

import type { ActiveTrail, GraphEdge } from '../../stores/radarStore';
import { getAgentColor } from '../../stores/radarStore';

// ───── Constants from 07-UI-SPEC.md ─────
export const COMET_TRAVEL_MS = 400;            // D-14: head animation duration
export const TRAIL_FULL_OPACITY_MS = 2000;     // D-16: opacity 100% phase
export const TRAIL_FADE_DURATION_MS = 8000;    // D-16: linear fade phase
export const TRAIL_TOTAL_LIFESPAN_MS = TRAIL_FULL_OPACITY_MS + TRAIL_FADE_DURATION_MS; // 10_000
export const MAX_TRAILS_PER_AGENT = 10;        // D-18
export const COMET_HEAD_RADIUS = 4;            // world-space, divide by zoom in canvas calls
export const COMET_HEAD_GLOW_RADIUS = 7;
export const COMET_TAIL_WIDTH_HEAD = 2;
export const COMET_TAIL_WIDTH_OLDEST = 0.5;
export const COMET_TAIL_SEGMENTS = 6;
export const AGENT_DOT_RADIUS = 6;
export const AGENT_PULSE_RING_1_MAX = 12;
export const AGENT_PULSE_RING_2_MAX = 20;
export const AGENT_PULSE_CYCLE_MS = 2000;
export const AGENT_PULSE_RING_2_DELAY_MS = 500;
export const AGENT_IDLE_MS = 30_000;          // UI-SPEC: stops pulsing when no events for 30s

// ───── Eased fraction (D-14: ease-out-cubic) ─────
export function easeOutCubic(t: number): number {
  const clamped = Math.max(0, Math.min(1, t));
  return 1 - Math.pow(1 - clamped, 3);
}

// ───── Head position interpolation ─────
export function interpolateHead(
  trail: ActiveTrail,
  now: number,
  fromPos: { x: number; y: number },
  toPos: { x: number; y: number },
): { x: number; y: number } {
  const t = (now - trail.startTs) / COMET_TRAVEL_MS;
  const eased = easeOutCubic(t);
  return {
    x: fromPos.x + (toPos.x - fromPos.x) * eased,
    y: fromPos.y + (toPos.y - fromPos.y) * eased,
  };
}

// ───── Trail opacity at age ms ─────
export function trailOpacity(ageMs: number): number {
  if (ageMs < 0) return 0;
  if (ageMs >= TRAIL_TOTAL_LIFESPAN_MS) return 0;
  if (ageMs < TRAIL_FULL_OPACITY_MS) return 1;
  const fadeProgress = (ageMs - TRAIL_FULL_OPACITY_MS) / TRAIL_FADE_DURATION_MS;
  return Math.max(0, 1 - fadeProgress);
}

// ───── Tail segment sampling (gradient stroke polyline) ─────
export interface TailSegment {
  x: number;
  y: number;
  alpha: number;
  width: number;
}

export function sampleTailSegments(
  trail: ActiveTrail,
  now: number,
  fromPos: { x: number; y: number },
  toPos: { x: number; y: number },
  segments: number = COMET_TAIL_SEGMENTS,
): TailSegment[] {
  const ageMs = now - trail.startTs;
  const baseAlpha = trailOpacity(ageMs);
  if (baseAlpha === 0) return [];
  const headT = Math.min(1, ageMs / COMET_TRAVEL_MS);
  const out: TailSegment[] = [];
  for (let i = 0; i < segments; i++) {
    // i = 0 is the oldest (tail end), i = segments-1 is the newest (head).
    const segT = (i / (segments - 1)) * easeOutCubic(headT);
    const x = fromPos.x + (toPos.x - fromPos.x) * segT;
    const y = fromPos.y + (toPos.y - fromPos.y) * segT;
    const segAlpha = baseAlpha * (i / (segments - 1));
    const segWidth = COMET_TAIL_WIDTH_OLDEST + (COMET_TAIL_WIDTH_HEAD - COMET_TAIL_WIDTH_OLDEST) * (i / (segments - 1));
    out.push({ x, y, alpha: segAlpha, width: segWidth });
  }
  return out;
}

// ───── Cull expired and FIFO-cap per agent ─────
export function cullExpiredTrails(
  trails: ActiveTrail[],
  now: number,
  capPerAgent: number = MAX_TRAILS_PER_AGENT,
  maxAgeMs: number = TRAIL_TOTAL_LIFESPAN_MS,
): ActiveTrail[] {
  const fresh = trails.filter(t => now - t.startTs < maxAgeMs);
  // Per-agent FIFO cap: keep latest N per agentId.
  const byAgent = new Map<string, ActiveTrail[]>();
  for (const t of fresh) {
    const arr = byAgent.get(t.agentId) ?? [];
    arr.push(t);
    byAgent.set(t.agentId, arr);
  }
  const out: ActiveTrail[] = [];
  for (const arr of byAgent.values()) {
    arr.sort((a, b) => a.startTs - b.startTs);  // oldest first
    out.push(...arr.slice(-capPerAgent));        // keep newest capPerAgent
  }
  // Stable sort by startTs for predictable render order
  out.sort((a, b) => a.startTs - b.startTs);
  return out;
}

// ───── drawCometTrails (UI-SPEC z-order steps 9-10) ─────
export function drawCometTrails(
  ctx: CanvasRenderingContext2D,
  trails: ActiveTrail[],
  positions: Map<string, { x: number; y: number }>,
  now: number,
  zoom: number,
): void {
  for (const trail of trails) {
    const fromPos = positions.get(trail.fromPath);
    const toPos = positions.get(trail.toPath);
    if (!fromPos || !toPos) continue;
    const color = getAgentColor(trail.agentId);

    // Tail: gradient stroke polyline (z-order step 9)
    const tail = sampleTailSegments(trail, now, fromPos, toPos);
    if (tail.length >= 2) {
      for (let i = 0; i < tail.length - 1; i++) {
        const seg = tail[i];
        const next = tail[i + 1];
        ctx.globalAlpha = (seg.alpha + next.alpha) / 2;
        ctx.strokeStyle = color;
        ctx.lineWidth = ((seg.width + next.width) / 2) / zoom;
        ctx.beginPath();
        ctx.moveTo(seg.x, seg.y);
        ctx.lineTo(next.x, next.y);
        ctx.stroke();
      }
      ctx.globalAlpha = 1;
    }

    // Head: filled circle + outer glow (z-order step 10), only while head is travelling
    const ageMs = now - trail.startTs;
    if (ageMs <= COMET_TRAVEL_MS) {
      const head = interpolateHead(trail, now, fromPos, toPos);
      const headOpacity = trailOpacity(ageMs);
      ctx.globalAlpha = headOpacity;
      // Outer glow
      const glowGrad = ctx.createRadialGradient(head.x, head.y, 0, head.x, head.y, COMET_HEAD_GLOW_RADIUS / zoom);
      glowGrad.addColorStop(0, `${color}66`);  // 40% alpha at center
      glowGrad.addColorStop(1, `${color}00`);
      ctx.fillStyle = glowGrad;
      ctx.beginPath();
      ctx.arc(head.x, head.y, COMET_HEAD_GLOW_RADIUS / zoom, 0, Math.PI * 2);
      ctx.fill();
      // Solid head
      ctx.fillStyle = color;
      ctx.beginPath();
      ctx.arc(head.x, head.y, COMET_HEAD_RADIUS / zoom, 0, Math.PI * 2);
      ctx.fill();
      ctx.globalAlpha = 1;
    }
  }
}

// ───── drawAgentDots (UI-SPEC z-order step 11) ─────
// agentDotMap: agentId → { x, y, lastEventTs }
export interface AgentDotState {
  agentId: string;
  x: number;
  y: number;
  lastEventTs: number;
}

export function drawAgentDots(
  ctx: CanvasRenderingContext2D,
  dots: AgentDotState[],
  now: number,
  zoom: number,
): void {
  for (const dot of dots) {
    const color = getAgentColor(dot.agentId);
    const idle = now - dot.lastEventTs > AGENT_IDLE_MS;
    if (!idle) {
      // Pulse rings (2s cycle, ring 2 delayed 0.5s) — UI-SPEC §Motion
      const cyclePhase = (now % AGENT_PULSE_CYCLE_MS) / AGENT_PULSE_CYCLE_MS;
      const cyclePhase2 = ((now + AGENT_PULSE_CYCLE_MS - AGENT_PULSE_RING_2_DELAY_MS) % AGENT_PULSE_CYCLE_MS) / AGENT_PULSE_CYCLE_MS;
      // Ring 1: 30%→0% opacity, scale 1→AGENT_PULSE_RING_1_MAX/AGENT_DOT_RADIUS
      ctx.strokeStyle = color;
      ctx.lineWidth = 1 / zoom;
      const r1 = (AGENT_DOT_RADIUS + (AGENT_PULSE_RING_1_MAX - AGENT_DOT_RADIUS) * cyclePhase) / zoom;
      ctx.globalAlpha = 0.3 * (1 - cyclePhase);
      ctx.beginPath();
      ctx.arc(dot.x, dot.y, r1, 0, Math.PI * 2);
      ctx.stroke();
      const r2 = (AGENT_DOT_RADIUS + (AGENT_PULSE_RING_2_MAX - AGENT_DOT_RADIUS) * cyclePhase2) / zoom;
      ctx.globalAlpha = 0.2 * (1 - cyclePhase2);
      ctx.beginPath();
      ctx.arc(dot.x, dot.y, r2, 0, Math.PI * 2);
      ctx.stroke();
      ctx.globalAlpha = 1;
    }
    // Center dot
    ctx.fillStyle = color;
    ctx.beginPath();
    ctx.arc(dot.x, dot.y, AGENT_DOT_RADIUS / zoom, 0, Math.PI * 2);
    ctx.fill();
  }
}

import { cullExpiredTrails, MAX_TRAILS_PER_AGENT, TRAIL_TOTAL_LIFESPAN_MS } from '../views/Radar/CometTrail';

// Inside RadarStore interface:
pushTrail: (t: ActiveTrail) => void;
pruneTrails: (now?: number) => void;

// Inside create<RadarStore>:
pushTrail: (t) => {
  set(s => {
    // FIFO cap: if agent at MAX_TRAILS_PER_AGENT, drop oldest for this agent
    const ofAgent = s.activeTrails.filter(x => x.agentId === t.agentId);
    let next: ActiveTrail[];
    if (ofAgent.length >= MAX_TRAILS_PER_AGENT) {
      const oldestId = ofAgent.sort((a, b) => a.startTs - b.startTs)[0].id;
      next = s.activeTrails.filter(x => x.id !== oldestId).concat(t);
    } else {
      next = s.activeTrails.concat(t);
    }
    return { activeTrails: next };
  });
},

pruneTrails: (now = Date.now()) => {
  set(s => ({ activeTrails: cullExpiredTrails(s.activeTrails, now, MAX_TRAILS_PER_AGENT, TRAIL_TOTAL_LIFESPAN_MS) }));
},

const events = usePipelineStore(s => s.events);
const lastAgentFileRef = useRef<Map<string, string>>(new Map()); // agentId → path
const agentDotsRef = useRef<Map<string, { x: number; y: number; lastEventTs: number }>>(new Map());

useEffect(() => {

  for (const ev of events) {

    const node = graphNodes.find(n => n.id === path);

        agentId,
        fromPath: prev,
        toPath: path,
        startTs: now,

    lastAgentFileRef.current.set(agentId, path);
