```tsx
// D-20. Renders graph extents bounding box scaled into 160x120px.
// PRESERVES commit e62272d: shifts right when manifest panel opens.
import { useEffect, useMemo, useRef } from 'react';
import { useRadarStore } from '../../stores/radarStore';
import { heatTintIfActive } from './HeatMapOverlay';

const MINIMAP_WIDTH = 160;
const MINIMAP_HEIGHT = 120;
const MINIMAP_PADDING = 2;
const MANIFEST_OPEN_RIGHT = 292;   // 280 panel + 12 gutter (UI-SPEC §Spacing Phase 7 Exceptions)
const MANIFEST_CLOSED_RIGHT = 12;
const NODE_DOT_SIZE = 2;            // CSS pixels (UI-SPEC §Sizing minimap node dot)
const VIEWPORT_STROKE = '#8eff71';  // primary (UI-SPEC §Color)

export function RadarMinimap({ canvasWidth, canvasHeight }: { canvasWidth: number; canvasHeight: number }) {
  const canvasRef = useRef<HTMLCanvasElement>(null);
  const graphNodes = useRadarStore(s => s.graphNodes);
  const viewport = useRadarStore(s => s.viewport);
  const isManifestOpen = useRadarStore(s => s.isManifestOpen);
  const heatMapEnabled = useRadarStore(s => s.heatMapEnabled);
  const contentionScores = useRadarStore(s => s.contentionScores);
  const setViewport = useRadarStore(s => s.setViewport);

  // Compute bounding box once per node mutation.
  const bbox = useMemo(() => {
    if (graphNodes.length === 0) return null;
    let minX = Infinity, minY = Infinity, maxX = -Infinity, maxY = -Infinity;
    for (const n of graphNodes) {
      if (n.x === undefined || n.y === undefined) continue;
      if (n.x < minX) minX = n.x;
      if (n.x > maxX) maxX = n.x;
      if (n.y < minY) minY = n.y;
      if (n.y > maxY) maxY = n.y;
    }
    if (minX === Infinity) return null;
    return { minX, minY, maxX, maxY, w: maxX - minX || 1, h: maxY - minY || 1 };
  }, [graphNodes]);

  useEffect(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;
    const dpr = window.devicePixelRatio || 1;
    canvas.width = MINIMAP_WIDTH * dpr;
    canvas.height = MINIMAP_HEIGHT * dpr;
    const ctx = canvas.getContext('2d')!;
    ctx.scale(dpr, dpr);
    ctx.fillStyle = '#201f1f';        // surface-container-high at 80% (UI-SPEC §Color)
    ctx.fillRect(0, 0, MINIMAP_WIDTH, MINIMAP_HEIGHT);

    if (bbox) {
      const innerW = MINIMAP_WIDTH - 2 * MINIMAP_PADDING;
      const innerH = MINIMAP_HEIGHT - 2 * MINIMAP_PADDING;
      const sx = innerW / bbox.w;
      const sy = innerH / bbox.h;

      // Draw nodes
      for (const n of graphNodes) {
        if (n.x === undefined || n.y === undefined) continue;
        const px = MINIMAP_PADDING + (n.x - bbox.minX) * sx;
        const py = MINIMAP_PADDING + (n.y - bbox.minY) * sy;
        const score = contentionScores.get(n.id) ?? 0;
        ctx.fillStyle = heatTintIfActive(score, heatMapEnabled);
        ctx.fillRect(px, py, NODE_DOT_SIZE, NODE_DOT_SIZE);
      }

      // Draw viewport rectangle
      // Main canvas world-coord visible region is canvas-screen-corners projected back to world.
      // viewport.zoom + viewport.panX/panY map: screenX = worldX * zoom + panX
      // So world rect: minWX = -panX/zoom, maxWX = (canvasWidth - panX) / zoom (similar for y).
      const wMinX = -viewport.panX / viewport.zoom;
      const wMaxX = (canvasWidth - viewport.panX) / viewport.zoom;
      const wMinY = -viewport.panY / viewport.zoom;
      const wMaxY = (canvasHeight - viewport.panY) / viewport.zoom;
      const rx = MINIMAP_PADDING + (wMinX - bbox.minX) * sx;
      const ry = MINIMAP_PADDING + (wMinY - bbox.minY) * sy;
      const rw = (wMaxX - wMinX) * sx;
      const rh = (wMaxY - wMinY) * sy;
      ctx.strokeStyle = VIEWPORT_STROKE;
      ctx.lineWidth = 1;
      ctx.strokeRect(rx, ry, rw, rh);
    }
  }, [bbox, graphNodes, viewport, isManifestOpen, heatMapEnabled, contentionScores, canvasWidth, canvasHeight]);

  const handleClick = (e: React.MouseEvent) => {
    if (!bbox) return;
    const rect = canvasRef.current!.getBoundingClientRect();
    const cx = e.clientX - rect.left;
    const cy = e.clientY - rect.top;
    const innerW = MINIMAP_WIDTH - 2 * MINIMAP_PADDING;
    const innerH = MINIMAP_HEIGHT - 2 * MINIMAP_PADDING;
    const worldX = bbox.minX + ((cx - MINIMAP_PADDING) / innerW) * bbox.w;
    const worldY = bbox.minY + ((cy - MINIMAP_PADDING) / innerH) * bbox.h;
    // Center the main viewport on (worldX, worldY)
    setViewport({
      panX: canvasWidth / 2 - worldX * viewport.zoom,
      panY: canvasHeight / 2 - worldY * viewport.zoom,
    });
  };

  return (
    <div
      data-testid="radar-minimap"
      style={{
        position: 'absolute',
        bottom: 12,
        right: isManifestOpen ? MANIFEST_OPEN_RIGHT : MANIFEST_CLOSED_RIGHT,
        width: MINIMAP_WIDTH,
        height: MINIMAP_HEIGHT,
        transition: 'right 200ms ease-in-out',
      }}
    >
      <canvas
        ref={canvasRef}
        onClick={handleClick}
        style={{
          width: MINIMAP_WIDTH,
          height: MINIMAP_HEIGHT,
          cursor: 'crosshair',
        }}
      />
    </div>
  );
}
```

**Step 1 — Conflict pulse + badge rendering.** Add to `src/views/Radar/RadarCanvas.tsx`:

```typescript
import { useConflictStore } from '../../stores/conflictStore';

const CONFLICT_PULSE_CYCLE_MS = 1600;
const CONFLICT_PULSE_INNER = 6;     // world-space (UI-SPEC §Sizing)
const CONFLICT_PULSE_OUTER = 15;
const CONFLICT_BADGE_SIZE = 4;
const CONFLICT_BADGE_OFFSET = 6;
const CONFLICT_COLOR = '#ff7351';   // error

const activeConflicts = useConflictStore(s => s.activeConflicts);

// In the render loop, after drawAgentDots and before ctx.restore():
function drawConflictPulses(
  ctx: CanvasRenderingContext2D,
  conflictedPaths: Set<string>,
  positions: Map<string, { x: number; y: number }>,
  now: number,
  zoom: number,
) {
  const cyclePhase = (now % CONFLICT_PULSE_CYCLE_MS) / CONFLICT_PULSE_CYCLE_MS;
  // Cubic-bezier(0,0,0.2,1) approximation: t * (2 - t) is close enough for opacity ramp
  const easedPhase = cyclePhase * (2 - cyclePhase);
  const r = (CONFLICT_PULSE_INNER + (CONFLICT_PULSE_OUTER - CONFLICT_PULSE_INNER) * cyclePhase) / zoom;
  const alpha = 1 - easedPhase;
  ctx.strokeStyle = CONFLICT_COLOR;
  ctx.lineWidth = 1 / zoom;
  for (const path of conflictedPaths) {
    const p = positions.get(path);
    if (!p) continue;
    ctx.globalAlpha = alpha;
    ctx.beginPath();
    ctx.arc(p.x, p.y, r, 0, Math.PI * 2);
    ctx.stroke();
  }
  ctx.globalAlpha = 1;
}

function drawConflictBadges(
  ctx: CanvasRenderingContext2D,
  conflictedPaths: Set<string>,
  positions: Map<string, { x: number; y: number }>,
  zoom: number,
) {
  ctx.fillStyle = CONFLICT_COLOR;
  for (const path of conflictedPaths) {
    const p = positions.get(path);
    if (!p) continue;
    ctx.beginPath();
    ctx.arc(p.x + CONFLICT_BADGE_OFFSET / zoom, p.y - CONFLICT_BADGE_OFFSET / zoom, CONFLICT_BADGE_SIZE / zoom, 0, Math.PI * 2);
    ctx.fill();
  }
}

// Inside the rAF callback, after drawAgentDots:
const conflictedPaths = useMemo(
  () => new Set(activeConflicts.map(c => c.filePath)),
  [activeConflicts],
);
// z-order step 12: pulse rings
drawConflictPulses(ctx, conflictedPaths, positions, now, viewport.zoom);
// z-order step 13: badge dots (always visible)
drawConflictBadges(ctx, conflictedPaths, positions, viewport.zoom);
```
