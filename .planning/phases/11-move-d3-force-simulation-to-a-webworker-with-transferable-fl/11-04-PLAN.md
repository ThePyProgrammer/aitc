    ```typescript
    // During active simulation, read world positions from the Worker-populated
    // Float32Array (ids[i] ↔ positions[i*2], positions[i*2+1]). When the Worker
    // has not yet emitted its first tick (positions.byteLength === 0) OR the
    // sim is idle, fall back to the store's graphNodes/positions.
    const simulating = isSimulatingRef.current;
    const live = simNodesRef.current;
    let liveNodes = s.graphNodes;
    let livePositions = s.positions;
    if (simulating && live.positions.byteLength > 0 && live.ids.length > 0) {
      simPositionMap.clear();
      // Repopulate simPositionMap from the Float32Array.
      for (let i = 0; i < live.ids.length; i++) {
        const id = live.ids[i];
        const x = live.positions[i * 2];
        const y = live.positions[i * 2 + 1];
        simPositionMap.set(id, { x, y });
      }
      livePositions = simPositionMap;
      // Build the scratch liveNodes array: merge Float32Array positions
      // with store-supplied metadata (dirKey/dirDepth) via the nodeById memo.
      // Allocate a fresh array each frame — with 5k nodes this is ~40KB of
      // short-lived memory which V8 handles trivially; the Map operations
      // dominate cost.
      const scratch: GraphNode[] = new Array(live.ids.length);
      for (let i = 0; i < live.ids.length; i++) {
        const id = live.ids[i];
        const meta = s.nodeById.get(id);
        if (!meta) {
          // Transient mismatch — a tick landed for a topology that the
          // store hasn't caught up to yet. Skip the frame's sim branch.
          scratch.length = 0;
          break;
        }
        scratch[i] = {
          ...meta,
          x: live.positions[i * 2],
          y: live.positions[i * 2 + 1],
        };
      }
      if (scratch.length === live.ids.length) {
        liveNodes = scratch;
      }
    }
    ```
