  //! Passive-scan bridge: ProcessSnapshot → AgentRegistry (AGNT-03, D-06, D-07).
  //!
  //! Spawned by start_watch, cancelled on stop_watch. Ticks every 2s, upserts
  //! PASSIVE-{pid} entries for candidates whose PID is not already owned by
  //! a non-PASSIVE registry entry, and reaps stale PASSIVE entries.

  use crate::agents::{AgentInfo, AgentRegistry, AgentState};
  use crate::agents::generic::passive_sentinel_adapter;
  use crate::pipeline::process_snapshot::ProcessSnapshot;
  use std::collections::HashSet;
  use std::sync::Arc;
  use std::time::Duration;
  use tokio::sync::RwLock;
  use tokio::task::JoinHandle;

  pub const BRIDGE_INTERVAL_MS: u64 = 2000;

  /// Spawn the passive-scan bridge task. Returns a JoinHandle the caller
  /// stores in ActiveWatch so it is aborted when the watch stops.
  pub fn spawn_passive_bridge(
      registry: Arc<AgentRegistry>,
      snapshot: Arc<RwLock<ProcessSnapshot>>,
      interval: Duration,
  ) -> JoinHandle<()> {
      tokio::spawn(async move {
          let mut tick = tokio::time::interval(interval);
          // Skip the immediate first tick so we don't double-fire with the refresher.
          tick.tick().await;
          loop {
              tick.tick().await;
              if let Err(e) = bridge_tick(&registry, &snapshot).await {
                  tracing::warn!(error = %e, "passive_bridge tick failed");
              }
          }
      })
  }

  /// One tick of the bridge. Public for direct unit testing.
  pub async fn bridge_tick(
      registry: &AgentRegistry,
      snapshot: &RwLock<ProcessSnapshot>,
  ) -> Result<(), String> {
      let candidates = {
          let snap = snapshot.read().await;
          snap.candidates()
      };
      let mut live_pids: HashSet<u32> = HashSet::with_capacity(candidates.len());
      for c in &candidates {
          live_pids.insert(c.pid);
      }
      // Reap first: drop PASSIVE entries whose PID is no longer live.
      registry.reap_passive_agents(&live_pids).await;

      // Upsert each live candidate as PASSIVE-{pid}, unless a non-passive entry
      // (e.g., KAGENT) already owns the PID (D-07).
      for c in candidates {
          if let Some(existing) = registry.find_agent_by_pid(c.pid).await {
              if !existing.id.starts_with("PASSIVE-") {
                  continue; // KAGENT or launched entry owns this PID; skip.
              }
          }
          let id = format!("PASSIVE-{}", c.pid);
          let info = AgentInfo {
              id: id.clone(),
              agent_type: "unknown".into(),
              protocol: "passive-scan".into(),
              state: AgentState::Running,
              pid: Some(c.pid),
              cwd: c.cwd.clone(),
              intent: None,
          };
          let adapter = passive_sentinel_adapter();
          if let Err(e) = registry.upsert_agent(id, info, adapter, false).await {
              tracing::warn!(pid = c.pid, error = %e, "passive upsert failed");
          }
      }
      Ok(())
  }

  /// Persist every attributed file event in `batch` to SQLite (D-09, HIST-01).
  /// Unattributed / Ambiguous events are silently skipped.
  pub(crate) async fn persist_attributed_batch(
      batch: &crate::pipeline::events::FileEventBatch,
      registry: &crate::agents::AgentRegistry,
      pool: &sqlx::SqlitePool,
  ) {
      use crate::pipeline::events::Attribution;
      for ev in &batch.events {
          let pid = match ev.attribution {
              Attribution::Pid(p) => p,
              _ => continue,
          };
          let Some(info) = registry.find_agent_by_pid(pid).await else {
              continue;
          };
          let session_id = match crate::db::session::ensure_open_session(
              &info.id,
              &info.agent_type,
              pool,
          )
          .await
          {
              Ok(id) => id,
              Err(e) => {
                  tracing::warn!(agent_id = %info.id, error = %e, "ensure_open_session failed in forwarder");
                  continue;
              }
          };
          let path = ev.path.to_string_lossy();
          if let Err(e) = crate::db::session::record_session_file_internal(
              session_id,
              &path,
              pool,
          )
          .await
          {
              tracing::warn!(session_id, path = %path, error = %e, "record_session_file_internal failed");
          }
      }
  }

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| OS process table → `ProcessSnapshot.candidates()` | PIDs and names come from the OS; allowlist filter already applied in Phase 2. |
| AgentRegistry ↔ forwarder | Forwarder reads `find_agent_by_pid`; a malicious process spoofing an allowlisted name could cause session attribution to the wrong agent_id. |
| Forwarder → SQLite | Every attributed batch writes to `session_files`; bounded by MAX_AGENTS and UNIQUE index. |

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-04-01 | Spoofing (malicious binary with allowlisted name) | passive_bridge | accept (Phase 2 inheritance) | Allowlist substring-match is a known Phase 2 risk (06-RESEARCH.md Security Domain). Passive entries have `agent_type: "unknown"` and no launch/terminate capability (passive_sentinel_adapter). Worst case: a rogue process appears as "unidentified" on the radar — view-only. |
