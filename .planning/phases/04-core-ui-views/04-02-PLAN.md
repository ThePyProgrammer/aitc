```typescript
import { create } from 'zustand';
import { invoke } from '@tauri-apps/api/core';
import { listen, type UnlistenFn } from '@tauri-apps/api/event';

export interface ApprovalRequest {
  id: number;
  agentId: string;
  requestType: string;
  filePath: string | null;
  diffContent: string | null;
  status: 'pending' | 'approved' | 'denied' | 'info_requested';
  urgency: 'low' | 'medium' | 'high';
  responseNote: string | null;
  editedContent: string | null;
  createdAt: string;
  resolvedAt: string | null;
}

export interface ChatMessage {
  id: number;
  agentId: string;
  direction: 'inbound' | 'outbound';
  content: string;
  deliveryStatus: 'delivered' | 'queued' | 'unsupported';
  approvalRequestId: number | null;
  createdAt: string;
}

interface CommsStore {
  requests: ApprovalRequest[];
  selectedRequestId: number | null;
  editingRequestId: number | null;  // frozen request during edit mode (D-05, Pitfall 3)
  messages: Record<string, ChatMessage[]>;
  isLoading: boolean;
  error: string | null;
  fetchRequests: () => Promise<void>;
  selectRequest: (id: number | null) => void;
  approveRequest: (id: number) => Promise<void>;
  denyRequest: (id: number) => Promise<void>;
  askMoreInfo: (id: number, question: string) => Promise<void>;
  approveWithEdits: (id: number, editedContent: string) => Promise<void>;
  setEditing: (id: number | null) => void;
  sendMessage: (agentId: string, content: string) => Promise<void>;
  fetchMessages: (agentId: string) => Promise<void>;
  subscribeToApprovals: () => Promise<UnlistenFn>;
  pendingCount: () => number;
  selectedRequest: () => ApprovalRequest | undefined;
  reset: () => void;
}
```

Key implementation details:
- `subscribeToApprovals`: use `listen('approval-request-created', ...)` pattern from conflictStore
- `setEditing(id)`: sets editingRequestId. When editingRequestId is set, `subscribeToApprovals` callback skips updating that specific request (Pitfall 3 from research)
- `pendingCount()`: `get().requests.filter(r => r.status === 'pending').length`
- `selectedRequest()`: `get().requests.find(r => r.id === get().selectedRequestId)`
- `sendMessage`: optimistic insert with deliveryStatus 'queued', then invoke, update status on success
- Error handling: wrap invoke calls in try/catch, set error state
