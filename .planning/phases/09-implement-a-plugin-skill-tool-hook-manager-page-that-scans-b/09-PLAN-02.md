       ```rust
       use std::collections::HashMap;
       use std::path::{Path, PathBuf};
       use std::sync::{Arc, RwLock};
       use std::time::{Duration, Instant};

       /// Default TTL — must exceed the 150ms debouncer window + a margin for
       /// tokio task scheduling delay. 2 seconds is the research-recommended
       /// value (Pitfall 3).
       const DEFAULT_TTL: Duration = Duration::from_secs(2);

       #[derive(Clone)]
       pub struct WriteFence {
           inner: Arc<RwLock<HashMap<PathBuf, Instant>>>,
           ttl: Duration,
       }

       impl WriteFence {
           pub fn new() -> Self { Self { inner: Arc::new(RwLock::new(HashMap::new())), ttl: DEFAULT_TTL } }
           #[cfg(test)]
           pub fn with_ttl(ttl: Duration) -> Self { Self { inner: Arc::new(RwLock::new(HashMap::new())), ttl } }
           pub fn record(&self, path: PathBuf) {
               let expires_at = Instant::now() + self.ttl;
               self.inner.write().expect("poisoned").insert(path, expires_at);
           }
           pub fn was_ours(&self, path: &Path) -> bool {
               match self.inner.read().expect("poisoned").get(path) {
                   Some(expiry) => Instant::now() < *expiry,
                   None => false,
               }
           }
           /// Drop all expired entries. Call periodically (e.g. once per batch).
           pub fn gc(&self) {
               let now = Instant::now();
               self.inner.write().expect("poisoned").retain(|_, exp| now < *exp);
           }
       }

       impl Default for WriteFence { fn default() -> Self { Self::new() } }
       ```
