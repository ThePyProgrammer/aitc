       ```rust
       /// Returns the two editable CLAUDE.md paths per D-13. `~/.claude/CLAUDE.md`
       /// is intentionally absent — global CLAUDE.md is read-only this phase.
       pub fn editable_paths(project_root: Option<&Path>) -> Vec<PathBuf> {
           let mut out = Vec::with_capacity(2);
           if let Some(p) = project_root {
               out.push(p.join("CLAUDE.md"));
               out.push(p.join(".claude").join("CLAUDE.md"));
           }
           out
       }

       pub fn is_editable(path: &Path, project_root: Option<&Path>) -> bool {
           editable_paths(project_root).iter().any(|p| p == path)
       }

       /// Atomically replace the contents of `path` with `content`. Uses
       /// tempfile::NamedTempFile::persist, which maps to ReplaceFile on
       /// Windows and rename(2) on Unix — both atomic at the kernel boundary.
       pub fn atomic_write(path: &Path, content: &str) -> Result<(), String> {
           let parent = path.parent().ok_or_else(|| "path has no parent".to_string())?;
           std::fs::create_dir_all(parent).map_err(|e| format!("mkdir parent: {e}"))?;
           let mut tmp = NamedTempFile::new_in(parent).map_err(|e| format!("tempfile: {e}"))?;
           tmp.write_all(content.as_bytes()).map_err(|e| format!("write_all: {e}"))?;
           tmp.flush().map_err(|e| format!("flush: {e}"))?;
           tmp.persist(path).map_err(|e| format!("persist: {}", e.error))?;
           Ok(())
       }
       ```

       ```rust
       /// Distinguishes an extra root's role for the single-Debouncer fan-out.
       /// `Global` = ~/.claude/, `Project` = <cwd>/.claude/.
       #[derive(Debug, Clone, Copy, PartialEq, Eq)]
       pub enum ScopeKind { Global, Project }

       impl From<ScopeKind> for crate::claude_resources::events::Scope {
           fn from(k: ScopeKind) -> Self {
               match k {
                   ScopeKind::Global => crate::claude_resources::events::Scope::Global,
                   ScopeKind::Project => crate::claude_resources::events::Scope::Project,
               }
           }
       }

       /// The allowlisted subdirs to watch under each extra root. Files outside
       /// these subdirs (plus settings.json and CLAUDE.md at the root) are NOT
       /// watched — Pitfall 1 (inotify allowlist).
       pub const EXTRA_ROOT_ALLOWLIST_SUBDIRS: &[&str] = &[
           "skills", "agents", "commands", "hooks", "plugins",
       ];

       /// Decide where a debounced event should route. Returns None if the path
       /// is outside every known root (drop). When an event targets a path
       /// under an extra root but outside the allowlist (e.g. ~/.claude/cache/),
       /// returns None. Used by process_debounce_result to fan out events.
       pub fn classify_event(
           path: &std::path::Path,
           repo_root: &std::path::Path,
           extra_roots: &[(std::path::PathBuf, ScopeKind)],
       ) -> Option<RoutedPath> {
           // Check extra roots first: a <cwd>/.claude/... path must route to
           // Resource, not Pipeline, even though <cwd> is also under repo_root.
           for (root, kind) in extra_roots {
               if path.starts_with(root) {
                   // Enforce allowlist + file-level matches for scope roots.
                   let suffix = match path.strip_prefix(root) {
                       Ok(s) => s,
                       Err(_) => continue,
                   };
                   let first = suffix.components().next().and_then(|c| c.as_os_str().to_str());
                   let is_allowed_subdir = first
                       .map(|f| EXTRA_ROOT_ALLOWLIST_SUBDIRS.contains(&f))
                       .unwrap_or(false);
                   let is_scope_level_file = matches!(
                       suffix.to_str(),
                       Some("settings.json") | Some("CLAUDE.md")
                   );
                   if is_excluded_subdir(first.unwrap_or("")) {
                       return None; // cache/, session-env/, etc.
                   }
                   if is_allowed_subdir || is_scope_level_file {
                       return Some(RoutedPath::Resource((*kind).into()));
                   }
                   return None;
               }
           }
           if path.starts_with(repo_root) {
               return Some(RoutedPath::Pipeline);
           }
           None
       }
       ```
