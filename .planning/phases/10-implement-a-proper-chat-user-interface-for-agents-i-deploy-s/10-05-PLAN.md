  <behavior>
fetchChannels:
- Calls `invoke<ChatChannel[]>('list_chat_channels')`; populates `channels` state.
- On error sets `error`.

loadInitialEvents(agentId):
- Calls `invoke<AgentEvent[]>('list_agent_events', { agentId, beforeId: null, limit: 50 })`.
- Sets `eventsByAgent[agentId]` to the returned array (reversed if backend returns newest-first — confirm based on Plan 02 SUMMARY; the ChatTranscript renders bottom-up so `eventsByAgent[agentId]` should store OLDEST→NEWEST; reverse the backend's newest-first list).

loadOlder(agentId):
- Finds the earliest event id in the current array.
- Calls `invoke<AgentEvent[]>('list_agent_events', { agentId, beforeId: <earliestId>, limit: 50 })`.
- PREPENDS results (in oldest-first order after reversing).

selectAgent(id):
- Sets `selectedAgentId`.
- If id is non-null, calls `markRead(id)` as a side effect.

sendMessage(agentId, content):
- Calls `invoke<AgentEvent>('send_chat_message_to_agent', { agentId, content })`.
- On success, optimistically appends the returned event to `eventsByAgent[agentId]` IF the `agent-event-appended` listener hasn't already (dedupe by event id).
- On error sets `error`.

clearThread(agentId):
- Calls `invoke<number>('clear_agent_thread', { agentId })`.
- Clears `eventsByAgent[agentId]` to `[]`.

markRead(agentId):
- Calls `invoke('mark_agent_events_read', { agentId })`.
- Resets `unreadByAgent[agentId] = 0`.

subscribeToChat:
- `Promise.all(6 listen calls)` — one per Tauri event name.
- `agent-event-appended` handler: append to `eventsByAgent[event.payload.agentId]` (dedupe by id); if `state.selectedAgentId !== event.payload.agentId || document.visibilityState !== 'visible'`, increment unreadByAgent[agentId].
- `agent-turn-started` handler: no state change; just logs (optional).
- `agent-turn-complete` handler: iterate `eventsByAgent[payload.agentId]` backwards to find the last `assistant_text` event with `streaming === true` and flip it to `streaming: false`. Also iterate to find the last `user_text` event and flip `deliveryStatus='consumed'`.
- `agent-session-started` handler: (optional) could insert a client-side session_boundary event, but Plan 02 already inserts one via the aggregator — the handler is a no-op.
- `agent-session-ended` handler: update the matching channel's `archived: true` in `channels[]`.
- `agent-delivery-updated` handler: find the event by `event.payload.eventId`, update its `deliveryStatus`.
- `agent-thread-cleared` handler: clear the matching `eventsByAgent[agentId]`.
- `agent-events-marked-read` handler: already reflected locally via markRead; this listener is defensive (handles backend-initiated resets).
- Returns a fan-out unlisten function.

totalUnread: simple reduce.

reset: zero everything.

useChatChannel hook: lightweight wrapper — subscribes on mount, unlistens on unmount. Also on mount, call `fetchChannels()` so the master list seeds on first render.

```typescript
import { create } from 'zustand';
import { invoke } from '@tauri-apps/api/core';
import { listen, type UnlistenFn } from '@tauri-apps/api/event';

export interface AgentEvent { /* (shape from Plan 01) */ }
export interface ChatChannel { /* (shape from Plan 01) */ }

type DeliveryStatus = 'queued' | 'delivered' | 'consumed' | 'unsupported';

interface DeliveryUpdatePayload { eventId: number; status: DeliveryStatus }
interface TurnCompletePayload { agentId: string; sessionId?: string | null; terminalReason: string; isError: boolean }
interface SessionEndedPayload { agentId: string; sessionId: string | null; reason: string; exitCode: number | null }

interface ChatStoreShape {
  eventsByAgent: Record<string, AgentEvent[]>;
  channels: ChatChannel[];
  selectedAgentId: string | null;
  unreadByAgent: Record<string, number>;
  archivedCollapsed: boolean;
  isLoading: boolean;
  error: string | null;
  // ... actions
}

export const useChatStore = create<ChatStoreShape>((set, get) => ({
  eventsByAgent: {},
  channels: [],
  selectedAgentId: null,
  unreadByAgent: {},
  archivedCollapsed: true,
  isLoading: false,
  error: null,

  fetchChannels: async () => {
    try {
      set({ isLoading: true, error: null });
      const channels = await invoke<ChatChannel[]>('list_chat_channels');
      set({ channels, isLoading: false });
    } catch (e) { set({ error: String(e), isLoading: false }); }
  },

  loadInitialEvents: async (agentId: string) => {
    try {
      const events = await invoke<AgentEvent[]>('list_agent_events', {
        agentId, beforeId: null, limit: 50,
      });
      // Backend returns newest-first; UI stores oldest-first.
      const oldestFirst = [...events].reverse();
      set((s) => ({ eventsByAgent: { ...s.eventsByAgent, [agentId]: oldestFirst } }));
    } catch (e) { set({ error: String(e) }); }
  },

  loadOlder: async (agentId: string) => {
    const current = get().eventsByAgent[agentId] ?? [];
    const beforeId = current[0]?.id ?? null;
    if (beforeId == null) return;
    try {
      const more = await invoke<AgentEvent[]>('list_agent_events', {
        agentId, beforeId, limit: 50,
      });
      if (more.length === 0) return;
      const oldestFirst = [...more].reverse();
      set((s) => ({
        eventsByAgent: { ...s.eventsByAgent, [agentId]: [...oldestFirst, ...current] },
      }));
    } catch (e) { set({ error: String(e) }); }
  },

  selectAgent: (id: string | null) => {
    set({ selectedAgentId: id });
    if (id != null) {
      void get().markRead(id);
    }
  },

  setArchivedCollapsed: (collapsed) => set({ archivedCollapsed: collapsed }),

  sendMessage: async (agentId: string, content: string) => {
    try {
      const event = await invoke<AgentEvent>('send_chat_message_to_agent', { agentId, content });
      set((s) => {
        const existing = s.eventsByAgent[agentId] ?? [];
        if (existing.some((e) => e.id === event.id)) return s;
        return { eventsByAgent: { ...s.eventsByAgent, [agentId]: [...existing, event] } };
      });
    } catch (e) { set({ error: String(e) }); }
  },

  clearThread: async (agentId: string) => {
    try {
      await invoke<number>('clear_agent_thread', { agentId });
      set((s) => ({ eventsByAgent: { ...s.eventsByAgent, [agentId]: [] } }));
    } catch (e) { set({ error: String(e) }); }
  },

  markRead: async (agentId: string) => {
    try {
      await invoke('mark_agent_events_read', { agentId });
      set((s) => ({ unreadByAgent: { ...s.unreadByAgent, [agentId]: 0 } }));
    } catch (e) { set({ error: String(e) }); }
  },

  subscribeToChat: async () => {
    const [un1, un2, un3, un4, un5, un6, un7, un8, un9] = await Promise.all([
      listen<AgentEvent>('agent-event-appended', (ev) => {
        const payload = ev.payload;
        const state = get();
        const existing = state.eventsByAgent[payload.agentId] ?? [];
        if (existing.some((e) => e.id === payload.id)) return;
        const isActiveAgent = state.selectedAgentId === payload.agentId;
        const isVisible = typeof document !== 'undefined' && document.visibilityState === 'visible';
        set({
          eventsByAgent: { ...state.eventsByAgent, [payload.agentId]: [...existing, payload] },
          unreadByAgent: isActiveAgent && isVisible
            ? state.unreadByAgent
            : { ...state.unreadByAgent, [payload.agentId]: (state.unreadByAgent[payload.agentId] ?? 0) + 1 },
        });
      }),
      listen<unknown>('agent-turn-started', () => { /* no-op */ }),
      listen<TurnCompletePayload>('agent-turn-complete', (ev) => {
        set((s) => {
          const arr = s.eventsByAgent[ev.payload.agentId] ?? [];
          let flippedStreaming = false;
          let flippedConsumed = false;
          const next = arr.map((e) => {
            if (!flippedStreaming && e.eventType === 'assistant_text' && (e.payloadJson as any)?.streaming) {
              flippedStreaming = true;
              return { ...e, payloadJson: { ...(e.payloadJson as object), streaming: false } };
            }
            return e;
          }).map((e) => {
            if (!flippedConsumed && e.eventType === 'user_text' && e.deliveryStatus === 'delivered') {
              flippedConsumed = true;
              return { ...e, deliveryStatus: 'consumed' as const };
            }
            return e;
          });
          return { eventsByAgent: { ...s.eventsByAgent, [ev.payload.agentId]: next } };
        });
      }),
      listen<unknown>('agent-session-started', () => { /* backend inserts session_boundary — no-op */ }),
      listen<SessionEndedPayload>('agent-session-ended', (ev) => {
        set((s) => ({
          channels: s.channels.map((c) =>
            c.agentId === ev.payload.agentId ? { ...c, archived: true } : c
          ),
        }));
      }),
      listen<DeliveryUpdatePayload>('agent-delivery-updated', (ev) => {
        const { eventId, status } = ev.payload;
        set((s) => {
          const nextMap: Record<string, AgentEvent[]> = {};
          for (const [aid, arr] of Object.entries(s.eventsByAgent)) {
            nextMap[aid] = arr.map((e) => (e.id === eventId ? { ...e, deliveryStatus: status } : e));
          }
          return { eventsByAgent: nextMap };
        });
      }),
      listen<string>('agent-thread-cleared', (ev) => {
        set((s) => ({ eventsByAgent: { ...s.eventsByAgent, [ev.payload]: [] } }));
      }),
      listen<string>('agent-events-marked-read', (ev) => {
        set((s) => ({ unreadByAgent: { ...s.unreadByAgent, [ev.payload]: 0 } }));
      }),
      listen<string>('agent-session-resumed', (ev) => {
        set((s) => ({
          channels: s.channels.map((c) =>
            c.agentId === ev.payload ? { ...c, archived: false } : c
          ),
        }));
      }),
    ]);
    return () => { un1(); un2(); un3(); un4(); un5(); un6(); un7(); un8(); un9(); };
  },

  totalUnread: () => Object.values(get().unreadByAgent).reduce((a, b) => a + b, 0),

  reset: () => set({
    eventsByAgent: {}, channels: [], selectedAgentId: null,
    unreadByAgent: {}, archivedCollapsed: true, isLoading: false, error: null,
  }),
}));
```

```typescript
import { useEffect, useRef } from 'react';
import { useChatStore } from '../stores/chatStore';

export function useChatChannel() {
  const unsubRef = useRef<(() => void) | null>(null);
  useEffect(() => {
    let cancelled = false;
    const chatStore = useChatStore.getState();
    chatStore.fetchChannels();
    chatStore.subscribeToChat().then((fn) => {
      if (cancelled) { fn(); return; }
      unsubRef.current = fn;
    });
    return () => {
      cancelled = true;
      unsubRef.current?.();
      unsubRef.current = null;
    };
  }, []);
}
```
