```rust
use chrono::Utc;
use serde::{Deserialize, Serialize};
use specta::Type;
use std::path::PathBuf;

#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize, Type)]
#[serde(tag = "kind", rename_all = "camelCase")]
pub enum FileEventKind {
    Create,
    Modify,
    Remove,
    Rename { from: PathBuf, to: PathBuf },
}

#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize, Type)]
#[serde(tag = "kind", content = "value", rename_all = "camelCase")]
pub enum Attribution {
    Pid(u32),
    Ambiguous(Vec<u32>),
    Unattributed,
}

#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize, Type)]
#[serde(rename_all = "camelCase")]
pub struct FileEvent {
    pub path: PathBuf,
    pub kind: FileEventKind,
    pub timestamp_ms: i64,
    pub attribution: Attribution,
}

impl FileEvent {
    pub fn new(path: PathBuf, kind: FileEventKind, attribution: Attribution) -> Self {
        Self {
            path,
            kind,
            timestamp_ms: Utc::now().timestamp_millis(),
            attribution,
        }
    }
}

#[derive(Debug, Clone, Serialize, Deserialize, Type)]
#[serde(rename_all = "camelCase")]
pub struct FileEventBatch {
    pub events: Vec<FileEvent>,
    pub batch_id: u64,
    pub dropped_batches: u32,
}

impl FileEventBatch {
    pub fn new_empty() -> Self {
        Self { events: Vec::new(), batch_id: 0, dropped_batches: 0 }
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn file_event_constructs_with_create_kind() {
        let ev = FileEvent::new(
            PathBuf::from("/tmp/foo.txt"),
            FileEventKind::Create,
            Attribution::Unattributed,
        );
        assert!(matches!(ev.kind, FileEventKind::Create));
        assert!(matches!(ev.attribution, Attribution::Unattributed));
    }

    #[test]
    fn empty_batch_has_zero_counts() {
        let b = FileEventBatch::new_empty();
        assert_eq!(b.events.len(), 0);
        assert_eq!(b.batch_id, 0);
        assert_eq!(b.dropped_batches, 0);
    }

    #[test]
    fn rename_event_roundtrips_json() {
        let ev = FileEvent::new(
            PathBuf::from("/tmp/a.txt"),
            FileEventKind::Rename { from: PathBuf::from("/tmp/a.txt"), to: PathBuf::from("/tmp/b.txt") },
            Attribution::Pid(1234),
        );
        let json = serde_json::to_string(&ev).unwrap();
        let back: FileEvent = serde_json::from_str(&json).unwrap();
        assert_eq!(ev, back);
    }

    #[test]
    fn attribution_pid_serializes_camelcase() {
        let a = Attribution::Pid(1234);
        let json = serde_json::to_string(&a).unwrap();
        assert!(json.contains("\"pid\""), "expected camelCase 'pid' tag, got: {json}");
        assert!(json.contains("1234"));
    }
}
```

```rust
//! Wave 0 smoke tests — answer the two research open questions before
//! downstream plans commit to the architecture.
//!
//! Open Question 1: Does tauri::ipc::Channel<T> outlive its registering command?
//! Open Question 2: What is sysinfo::refresh_processes_specifics cost on this
//!                  machine? Target <50ms for 1000ms polling cadence.
//!
//! These tests do NOT require a Tauri runtime; they exercise the primitives
//! directly. A full end-to-end Channel test inside a Tauri command happens in
//! Plan 04 once the command exists.

use crate::pipeline::events::{Attribution, FileEvent, FileEventBatch, FileEventKind};
use std::path::PathBuf;
use std::time::Instant;

/// Smoke test #1: verify that a tauri::ipc::Channel<T> can be constructed, cloned,
/// stored in app state (Arc), and used to .send() after the "registering command"
/// scope has ended. We simulate the lifetime by dropping the initial owning scope
/// and sending from a later scope.
///
/// NOTE: tauri::ipc::Channel requires an ipc::InvokeResponder to construct in
/// production. In this unit test we exercise the Clone/Send/Sync contract via the
/// type's public API surface. The full runtime test (inside an actual Tauri
/// command) is in Plan 04 Task 2.
///
/// What this test DOES prove:
///   - Channel<FileEventBatch> implements Clone + Send + Sync + 'static
///   - FileEventBatch can be passed through the channel's serde boundary
///
/// What this test does NOT prove (deferred to Plan 04 runtime test):
///   - Whether the channel errors or silently drops when the webview unmounts
#[test]
fn channel_type_is_clone_send_sync_static() {
    fn assert_clone<T: Clone>() {}
    fn assert_send<T: Send>() {}
    fn assert_sync<T: Sync>() {}
    fn assert_static<T: 'static>() {}

    assert_clone::<tauri::ipc::Channel<FileEventBatch>>();
    assert_send::<tauri::ipc::Channel<FileEventBatch>>();
    assert_sync::<tauri::ipc::Channel<FileEventBatch>>();
    assert_static::<tauri::ipc::Channel<FileEventBatch>>();
}

/// Smoke test #2: FileEventBatch serializes cleanly for Channel transport.
/// A Channel<T> calls serde_json serialization under the hood when sending to
/// the frontend. This test proves our batch type is serializable.
#[test]
fn file_event_batch_serializes_for_channel_transport() {
    let batch = FileEventBatch {
        events: vec![
            FileEvent::new(
                PathBuf::from("C:\\Users\\test\\repo\\src\\main.rs"),
                FileEventKind::Modify,
                Attribution::Pid(1234),
            ),
            FileEvent::new(
                PathBuf::from("C:\\Users\\test\\repo\\README.md"),
                FileEventKind::Create,
                Attribution::Unattributed,
            ),
        ],
        batch_id: 42,
        dropped_batches: 0,
    };
    let json = serde_json::to_string(&batch).expect("serialize");
    assert!(json.contains("\"batchId\":42"));
    assert!(json.contains("\"droppedBatches\":0"));
    assert!(json.contains("\"events\""));
}

/// Benchmark #1: sysinfo::System::refresh_processes_specifics cost on this box.
/// Target: <50ms per refresh to justify a 1000ms polling cadence.
///
/// Marked #[ignore] so it runs only on-demand via `cargo test -- --ignored`.
/// The benchmark prints its result to stdout via println!. A Wave 0 run of
/// `cargo test pipeline::smoke_tests::bench_sysinfo_refresh_cost -- --ignored --nocapture`
/// captures the number.
///
/// Research finding (02-RESEARCH.md Pitfall 4): sysinfo's Windows PEB reads
/// cost 30-100ms on a typical dev box with ~300 processes. If this benchmark
/// reports >50ms, downstream Plan 03 must either reduce refresh scope via
/// ProcessRefreshKind or increase the polling tick to 2000ms.
#[test]
#[ignore]
fn bench_sysinfo_refresh_cost() {
    use sysinfo::{ProcessRefreshKind, ProcessesToUpdate, System, UpdateKind};

    let mut sys = System::new();
    // Warm-up refresh (first call is slower due to initial allocations)
    sys.refresh_processes_specifics(
        ProcessesToUpdate::All,
        true,
        ProcessRefreshKind::nothing()
            .with_cwd(UpdateKind::Always)
            .with_cmd(UpdateKind::OnlyIfNotSet)
            .with_exe(UpdateKind::OnlyIfNotSet),
    );

    // Measured runs
    let mut samples = Vec::with_capacity(5);
    for _ in 0..5 {
        let t0 = Instant::now();
        sys.refresh_processes_specifics(
            ProcessesToUpdate::All,
            true,
            ProcessRefreshKind::nothing()
                .with_cwd(UpdateKind::Always)
                .with_cmd(UpdateKind::OnlyIfNotSet)
                .with_exe(UpdateKind::OnlyIfNotSet),
        );
        samples.push(t0.elapsed().as_millis());
    }
    let avg_ms: u128 = samples.iter().sum::<u128>() / samples.len() as u128;
    let process_count = sys.processes().len();

    println!(
        "bench_sysinfo_refresh_cost: {} processes, avg={}ms, samples={:?}",
        process_count, avg_ms, samples
    );

    // Fail only if WILDLY over target (10x). At or above 500ms, the 1000ms
    // polling cadence would peg a CPU. Below 500ms, Plan 03 can proceed even if
    // 50ms target is exceeded (cadence may need tuning).
    assert!(
        avg_ms < 500,
        "sysinfo refresh averaged {}ms over {} samples — this will peg a CPU at 1Hz polling. \
         Plan 03 must either (a) narrow ProcessRefreshKind, (b) pre-filter by name, or \
         (c) increase polling tick. See 02-RESEARCH.md Pitfall 4.",
        avg_ms, samples.len()
    );

    if avg_ms >= 50 {
        println!(
            "WARNING: sysinfo refresh averaged {}ms > 50ms target. \
             Plan 03 should document the measured value and consider polling at 2000ms.",
            avg_ms
        );
    }
}
```
