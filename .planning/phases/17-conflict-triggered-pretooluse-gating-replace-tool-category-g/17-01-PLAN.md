    ```rust
    //! safelist for the conflict-gate predicate. CONTEXT D-09..D-13.

    //! before parsing so common read-only tools stay zero-overhead.

        Safelisted,

        ParseFailed,

        // Cheap pre-check: any stdout-ish redirect forbids safelist (D-11 last bullet).
        let has_redirect = command.contains('>');

                tracing::debug!(
                    kind = "bash_parse",
                    command_len = command.len(),
                    tokens = 0_usize,
                    result = "ParseFailed",
                    "shlex::split returned None or empty"

                return BashParseResult::ParseFailed;
    ```

    ```rust
    fn split_on_operators(argv: Vec<String>, ops: &[&str]) -> Vec<Vec<String>> {
        let mut segments = Vec::new();
        let mut current = Vec::new();
        for tok in argv {
            if ops.contains(&tok.as_str()) {
                if !current.is_empty() {
                    segments.push(std::mem::take(&mut current));
                }
            } else {
                current.push(tok);
            }
        }
        if !current.is_empty() {
            segments.push(current);
        }
        segments
    }

    fn resolve(cwd: &Path, raw: &str) -> PathBuf {
        let p = Path::new(raw);
        if p.is_absolute() { p.to_path_buf() } else { cwd.join(p) }
    }

    fn parse_one_segment(segment: &[String], cwd: &Path) -> Vec<PathBuf> {
        if segment.is_empty() { return Vec::new(); }
        let first = segment[0].as_str();
        let mut targets = Vec::new();

        // Stdout/stderr redirect verbs — look for `>`, `>>`, `2>`, `&>` tokens with next arg.
        // shlex returns them as independent tokens.
        let mut i = 0;
        while i < segment.len() {
            let t = segment[i].as_str();
            if matches!(t, ">" | ">>" | "2>" | "&>") && i + 1 < segment.len() {
                targets.push(resolve(cwd, &segment[i + 1]));
                i += 2;
                continue;
            }
            i += 1;
        }

        // Verb dispatch (D-12). Runs REGARDLESS of redirect-token presence so
        // "cp a b > log" yields [cwd/b, cwd/log].
        match first {
            "cp" | "mv" | "install" => {
                // Last positional arg is dst. Skip flags (starts with '-').
                let positionals: Vec<&String> = segment.iter()
                    .skip(1).filter(|t| !t.starts_with('-')).collect();
                if let Some(dst) = positionals.last() {
                    targets.push(resolve(cwd, dst));
                }
            }
            "rm" | "touch" | "mkdir" | "patch" => {
                // All positional path args are targets.
                for t in segment.iter().skip(1) {
                    if !t.starts_with('-') && !matches!(t.as_str(), ">" | ">>" | "2>" | "&>") {
                        targets.push(resolve(cwd, t));
                    }
                }
            }
            "sed" | "awk" => {
                // `sed -i <expr> <path>…` / `awk -i inplace <prog> <path>…` — last positional
                // is the path. For awk, only treat as write target when `-i inplace` present.
                let has_inplace = segment.iter().any(|t| t == "-i"
                    || (first == "awk" && segment.windows(2).any(|w| w[0] == "-i" && w[1] == "inplace")));
                if first == "sed" && has_inplace {
                    if let Some(last) = segment.iter().rev()
                        .find(|t| !t.starts_with('-') && !matches!(t.as_str(), ">" | ">>" | "2>" | "&>"))
                    {
                        if last != &segment[0] {
                            targets.push(resolve(cwd, last));
                        }
                    }
                }
                if first == "awk" && segment.iter().any(|t| t == "-i")
                    && segment.iter().any(|t| t == "inplace")
                {
                    if let Some(last) = segment.iter().rev()
                        .find(|t| !t.starts_with('-') && t != &"inplace" && !matches!(t.as_str(), ">" | ">>" | "2>" | "&>"))
                    {
                        if last != &segment[0] {
                            targets.push(resolve(cwd, last));
                        }
                    }
                }
            }
            "dd" => {
                // `dd of=PATH`.
                for t in segment.iter().skip(1) {
                    if let Some(path) = t.strip_prefix("of=") {
                        targets.push(resolve(cwd, path));
                    }
                }
            }
            "tee" => {
                // `tee [-a] PATH…` — all non-flag args are targets.
                for t in segment.iter().skip(1) {
                    if !t.starts_with('-') {
                        targets.push(resolve(cwd, t));
                    }
                }
            }
            _ => { /* unknown verb → only redirect-derived targets (if any) */ }
        }

        targets
    }
    ```
