Cross-crate binary invocation: integration tests in the `aitc` crate (under `src-tauri/tests/`) cannot use `env!("CARGO_BIN_EXE_aitc-hook")` directly (that env var is only set for the crate that owns the binary). Two options:

1. Move the new e2e tests into `src-tauri/aitc-hook/tests/`, and factor out the `start_registration_server` + DB setup into a `pub` helper in a new `src-tauri/src/testing.rs` module behind `#[cfg(any(test, feature = "test-support"))]`. Then `aitc-hook/tests/*.rs` imports from `aitc` with a dev-dependency.

2. Use `cargo metadata` at test-time to locate `target/debug/aitc-hook{,.exe}` and run it directly.

Option 2 is simpler and does not require feature flags. Use it:

```rust
// src-tauri/tests/hook_e2e_with_real_sidecar.rs
//! End-to-end Phase 8 integration: real compiled aitc-hook binary driving the
//! real start_registration_server + real SQLite + real AgentRegistry + real
//! WaiterRegistry. Tests the whole stack except Claude Code itself.

use std::io::Write;
use std::process::{Command, Stdio};
use std::sync::Arc;
use std::time::Duration;

use sqlx::sqlite::SqlitePoolOptions;
use serde_json::{json, Value};

use aitc::agents::hook_waiters::{HookDecision, WaiterRegistry};
use aitc::agents::registry::AgentRegistry;
use aitc::agents::adapter::{AgentInfo, AgentState};
use aitc::agents::self_register::start_registration_server;

fn sidecar_binary_path() -> std::path::PathBuf {
    // Cargo workspace layout: sidecar binary lives in target/{profile}/aitc-hook{.exe}.
    // Integration tests run from the crate root, so target/ is at ../target/.
    let mut p = std::env::current_exe().unwrap();
    // current_exe is something like .../target/debug/deps/hook_e2e_with_real_sidecar-<hash>
    while !p.ends_with("target") {
        if !p.pop() { panic!("failed to locate target/ dir"); }
    }
    let profile = if cfg!(debug_assertions) { "debug" } else { "release" };
    let mut bin = p.join(profile).join("aitc-hook");
    if cfg!(target_os = "windows") { bin.set_extension("exe"); }
    assert!(bin.exists(), "sidecar binary missing at {bin:?}; run `cargo build -p aitc-hook` first");
    bin
}

fn fixture() -> String {
    std::fs::read_to_string(
        std::path::Path::new(env!("CARGO_MANIFEST_DIR")).join("tests/fixtures/pretool_use_stdin.json")
    ).unwrap()
}

async fn setup_server() -> (u16, Arc<WaiterRegistry>, sqlx::SqlitePool, u32) {
    let pool = SqlitePoolOptions::new().max_connections(5)
        .connect("sqlite::memory:").await.unwrap();
    // Apply migrations (same path the app uses).
    sqlx::migrate!("./src/db/migrations").run(&pool).await.unwrap();

    let mut reg = AgentRegistry::new();
    reg.register_adapter(Arc::new(aitc::agents::claude_code::ClaudeCodeAdapter));
    let registry = Arc::new(reg);

    // Spawn a long-lived subprocess to use as the agent PID.
    let sleep_child = if cfg!(target_os = "windows") {
        Command::new("ping").args(["127.0.0.1", "-n", "60"]).spawn().unwrap()
    } else {
        Command::new("sleep").args(["60"]).spawn().unwrap()
    };
    let live_pid = sleep_child.id();
    // Leak the child — test ends before it finishes.
    std::mem::forget(sleep_child);

    registry.upsert_agent(
        format!("KAGENT-{live_pid}"),
        AgentInfo {
            id: format!("KAGENT-{live_pid}"),
            agent_type: "claude-code".into(),
            protocol: "http".into(),
            state: AgentState::Running,
            pid: Some(live_pid),
            cwd: Some(std::path::PathBuf::from("/home/dev/proj")),
            intent: None,
        },
        registry.find_adapter_for_process("claude-code").unwrap(),
        false,
    ).await.unwrap();

    let waiters = WaiterRegistry::new_arc();
    let app = tauri::test::mock_app();
    let port = start_registration_server(
        registry.clone(), pool.clone(), waiters.clone(), app.handle().clone(), 0
    ).await.unwrap();
    (port, waiters, pool, live_pid)
}

async fn pending_row_id(pool: &sqlx::SqlitePool) -> Option<i64> {
    // Poll for up to 2s for a pretool_use row to appear.
    for _ in 0..40 {
        let row: Option<(i64,)> = sqlx::query_as(
            "SELECT id FROM approval_requests WHERE request_type = 'pretool_use' AND status = 'pending' ORDER BY id DESC LIMIT 1"
        ).fetch_optional(pool).await.unwrap_or(None);
        if let Some((id,)) = row { return Some(id); }
        tokio::time::sleep(Duration::from_millis(50)).await;
    }
    None
}

#[tokio::test(flavor = "multi_thread", worker_threads = 3)]
async fn e2e_allow_roundtrip_with_real_sidecar() {
    let (port, waiters, pool, _) = setup_server().await;

    // Start sidecar in a blocking thread (std::process::Command is sync).
    let sidecar = sidecar_binary_path();
    let fixture_bytes = fixture();
    let sidecar_handle = tokio::task::spawn_blocking(move || {
        let mut child = Command::new(sidecar)
            .env("AITC_PORT", port.to_string())
            .env_remove("AITC_PORT_FILE_OVERRIDE")
            .stdin(Stdio::piped()).stdout(Stdio::piped()).stderr(Stdio::piped())
            .spawn().unwrap();
        child.stdin.as_mut().unwrap().write_all(fixture_bytes.as_bytes()).unwrap();
        child.wait_with_output().unwrap()
    });

    // Wait for the pretool_use row to land, then signal allow.
    let row_id = pending_row_id(&pool).await.expect("pretool_use row not created");
    assert!(waiters.signal(row_id, HookDecision::Allow).await);

    let out = sidecar_handle.await.unwrap();
    assert_eq!(out.status.code(), Some(0),
               "sidecar stderr: {}", String::from_utf8_lossy(&out.stderr));
    let parsed: Value = serde_json::from_slice(&out.stdout).unwrap();
    assert_eq!(parsed["hookSpecificOutput"]["permissionDecision"], "allow");
    assert_eq!(parsed["hookSpecificOutput"]["hookEventName"], "PreToolUse");
    assert!(parsed.get("decision").is_none(), "must not emit deprecated envelope");
}

async fn e2e_allow_with_edits_roundtrip_with_real_sidecar() {

#[tokio::test(flavor = "multi_thread", worker_threads = 3)]
async fn e2e_deny_roundtrip_with_real_sidecar() {
    let (port, waiters, pool, _) = setup_server().await;
    let sidecar = sidecar_binary_path();
    let fixture_bytes = fixture();
    let sidecar_handle = tokio::task::spawn_blocking(move || {
        let mut child = Command::new(sidecar)
            .env("AITC_PORT", port.to_string())
            .env_remove("AITC_PORT_FILE_OVERRIDE")
            .stdin(Stdio::piped()).stdout(Stdio::piped()).stderr(Stdio::piped())
            .spawn().unwrap();
        child.stdin.as_mut().unwrap().write_all(fixture_bytes.as_bytes()).unwrap();
        child.wait_with_output().unwrap()

    assert_eq!(out.status.code(), Some(2));
    assert!(out.stdout.is_empty(), "deny path must not write stdout JSON");
    assert!(String::from_utf8_lossy(&out.stderr).contains("user rejected"));
}

#[tokio::test(flavor = "multi_thread", worker_threads = 3)]
async fn e2e_abandon_when_sidecar_killed() {
    let (port, _waiters, pool, _) = setup_server().await;
    let sidecar = sidecar_binary_path();
    let fixture_bytes = fixture();
    // Spawn sidecar but kill it ~100ms in (simulates Claude dying while blocked).
    let mut child = Command::new(&sidecar)
        .env("AITC_PORT", port.to_string())
        .env_remove("AITC_PORT_FILE_OVERRIDE")
        .stdin(Stdio::piped()).stdout(Stdio::piped()).stderr(Stdio::piped())
        .spawn().unwrap();
    child.stdin.as_mut().unwrap().write_all(fixture_bytes.as_bytes()).unwrap();
    // Wait for row creation, then kill.
    let row_id = pending_row_id(&pool).await.unwrap();
    child.kill().unwrap();
    let _ = child.wait();

    // Give the AbandonGuard up to 2s to fire.
    let mut status = String::new();
    for _ in 0..40 {
        let r: Option<(String,)> = sqlx::query_as(
            "SELECT status FROM approval_requests WHERE id = ?"
        ).bind(row_id).fetch_optional(&pool).await.unwrap();
        if let Some((s,)) = r {
            status = s;
            if status == "abandoned" { break; }
        }
        tokio::time::sleep(Duration::from_millis(50)).await;
    }
    assert_eq!(status, "abandoned", "expected AbandonGuard to mark row abandoned");
}
```
