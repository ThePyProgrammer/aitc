       ```rust
       /// Serialize the in-memory tree index into repo-relative TreeIndexEntry rows.
       /// Skips the repo-root entry itself (frontend synthesizes the root node).
       /// Normalizes Windows backslashes to forward slashes so the frontend
       /// treemap's `split('/')` yields correct segments.
       pub(crate) fn serialize_tree_index(
           tree: &HashMap<PathBuf, FileNode>,
           repo_root: &Path,
       ) -> Vec<TreeIndexEntry> {
           tree.iter()
               .filter_map(|(path, node)| {
                   let rel = path.strip_prefix(repo_root).ok()?;
                   // Skip the repo-root entry itself (empty relative path).
                   if rel.as_os_str().is_empty() {
                       return None;
                   }
                   let path_str = rel.to_string_lossy().replace('\\', "/");
                   let depth = rel.components().count() as u32;
                   Some(TreeIndexEntry {
                       path: path_str,
                       size: node.size,
                       is_dir: node.is_dir,
                       depth,
                   })
               })
               .collect()
       }
       ```
       Then replace the body of `get_tree_index` with a call to `serialize_tree_index(&active.tree_index, &active.repo_root)`.

    4. Do NOT alter `tree_index.rs` — HashMap keys remain absolute paths (attribution, reconciliation, and conflict detection depend on that invariant).
